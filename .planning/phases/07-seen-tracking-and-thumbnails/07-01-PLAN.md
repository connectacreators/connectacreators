--- FIX 3 (THUMB-01, THUMB-02, THUMB-03): ViralToday.tsx — gradient fallback in VideoCard ---

ViralToday.tsx does NOT have a `gradientFor` function or `PALETTES` constant. Add them near the top of the file, after the existing `PLATFORM_ICON` constant (around line 318). Copy the exact implementation from ViralReelFeed.tsx:

```javascript
const GRID_PALETTES = [
  ["#0f0c1e", "#3d1054"], ["#001624", "#003d5c"],
  ["#0a0a14", "#1a3a5c"], ["#0c0c0c", "#1a0a2e"],
  ["#001a10", "#003320"], ["#1a001a", "#3d0066"],
  ["#1a0a00", "#3d2000"], ["#000d1a", "#001f3d"],
];

function gridGradientFor(name: string) {
  let h = 0;
  for (const c of name || "") h = (h * 31 + c.charCodeAt(0)) & 0xffff;
  const p = GRID_PALETTES[h % GRID_PALETTES.length];
  return `linear-gradient(160deg, ${p[0]} 0%, ${p[1]} 100%)`;
}
```
